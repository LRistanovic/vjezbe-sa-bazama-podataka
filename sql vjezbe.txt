SELECT film.naslov, COUNT(grad.naziv) FROM
film JOIN projekcija
ON film.fid = projekcija.fid
JOIN bioskop
ON projekcija.bid = bioskop.bid
JOIN grad
ON bioskop.naziv = grad.naziv
GROUP BY film.fid
WHERE COUNT(grad.naziv) > 5

b)
SELECT film.fid, COUNT(projekcija.pid) FROM
film JOIN projekcija
ON film.fid = projekcija.fid
GROUP BY film.fid
ORDER BY COUNT(projekcija.pid) DESC
LIMIT 1
